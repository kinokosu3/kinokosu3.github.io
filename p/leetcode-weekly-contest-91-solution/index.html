class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {

        int n = A.size(); //è¡Œ 3
        int m = A[0].size(); //åˆ— 4
        for(int i=0;i<n;i++){
            if(A[i][0] == 0){
                for(int j=0;j<m;j++) A[i][j] = 1 - A[i][j];
            }
        }

        int ans = 0;
        for(int j=0;j<m;j++){
            int p = m - j - 1;
            int ocnt = 0, zcnt = 0;
            for(int i=0;i<n;i++){
                if(A[i][j] == 1) ocnt++;
                else zcnt++;
            }
            ans += max(ocnt, zcnt) * (1 << p);
        }
        return ans;
    }
};
